### baekjoon 1303 bfs 문제.   

- 기본적인 bfs & dfs 문제이다.   

- dfs로 풀 경우 각 팀별로 구분해야 되는 번거로움이 있기에,   
 bfs로 풀었다.   


```python
import sys
from collections import deque

a,b = 0, 0
N, M = map(int, sys.stdin.readline().split())

temp = [list(sys.stdin.readline().rstrip()) for _ in range(M)]

dx = [1, -1, 0, 0]
dy = [0, 0, 1, -1]

def bfs(x,y, bw):
    queue = deque()
    queue.append((x,y))
    cnt = 0
    temp[x][y] = 0
    while queue:
        x, y = queue.popleft()
        cnt += 1

        for i in range(4):
            # 상하좌우 다 돌아다니면서 순회
            nx = x + dx[i]
            ny = y + dy[i]
            if nx < 0 or ny <0 or nx >= M or ny >= N:
                continue

            if temp[nx][ny] != 0 and temp[nx][ny] == bw:
                queue.append((nx,ny))
                temp[nx][ny] = 0
        
    return (1 if cnt == 0 else cnt)


for i in range(M):
    for j in range(N):
        if temp[i][j] == "W":
            a += bfs(i,j,temp[i][j])**2
        elif temp[i][j] == "B":
            b += bfs(i,j,temp[i][j])**2
print(a,b)



```
